Micro-modeling (Solution 1)

F <- c(0,5,0,
       0,0,1,
       0,0,0)
F <- matrix(F,nrow=3,byrow=TRUE)
z <- c(10,0,0)
e <- c(3,3,3)
r <-c(1,1,1)
X <- c(10,3,1)
liv <- c(TRUE,TRUE,FALSE)
toy <- pack(F,z,e,r,X,liv)
enaFlow(toy)
